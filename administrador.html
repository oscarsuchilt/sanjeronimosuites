Configuración de Supabase

SQL para crear la tabla en Supabase

CREATE TABLE suites (
  id TEXT PRIMARY KEY,
  nombre TEXT NOT NULL,
  tipo TEXT NOT NULL,
  piso INTEGER NOT NULL,
  descripcion TEXT,
  tags TEXT[],
  precio_noche INTEGER,
  precio_semana INTEGER,
  precio_mes INTEGER,
  airbnb_url TEXT,
  fotos TEXT[],
  activa BOOLEAN DEFAULT true,
  orden INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Storage bucket: crear "fotos" como público
-- En Supabase → Storage → New bucket → "fotos" → Public
